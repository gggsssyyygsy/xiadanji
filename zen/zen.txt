  (function($) {
    $.extend({
      'cartPopupIntervals': [],
      'renderCartPopup': function(event){
        $('.shopping-cart-small').css('display', '');
        $('.shopping-cart-small').addClass('hover');
        var url = '/shopping-cart?small=true';
        if(event.target == $('.shopping-cart-small .more a')[0]) url += '&all=true'
        return $.ajax({
          url: url,
        })
        .then(function(content){
          var badgeOpen = $('.shopping-cart-small').html(content);
          var cardCount = badgeOpen.find('.card-count').text();
          $('.shopping-cart .badge').text(cardCount);
          if(cardCount > 0) $('.shopping-cart .badge').removeClass('hidden');
          $('.shopping-cart-small .more a').on('click', function(event){
            $.renderCartPopup(event);
          });
          var flashed = 0;
          $.each($.cartPopupIntervals, function(){clearInterval(this);});
          $.cartPopupIntervals = [];
          $('.popup-cart-verified-spinner').each(function(index, elem) {
            new Spinner({scale: 0.35, top: '-10px', left: '100px'}).spin(elem)
          });
          $('[data-toggle="tooltip"]').tooltip();
          if($(event.originalEvent.target).hasClass('add-to-cart')){
            interval = setInterval(function(){
              if(flashed > 4) clearInterval(interval);
              $('.shopping-cart-small .new-item, .shopping-cart-small .subtotal .overflow-h').toggleClass('highlighted');
              flashed++;
            },250);
            $.cartPopupIntervals.push(interval);
            badgeOpen.delay(3000).queue(function() {
              $(this).removeClass('hover').dequeue();
            });
          }
        }, function(xhr, status, error){
          $('.shopping-cart-small')
            .html($('<li />')
            .text(status + ': ' + error))
            .addClass('hover');
        });
      }
    });
  
    $(function(){
      $('#open-cart').mouseover(function(e) {
        $.renderCartPopup(e);
      });
      $('.shopping-cart-small').mouseover(function(e) {
        $(this).addClass('hover');
      });
      $('.shopping-cart-small').mouseleave(function(e) {
        $(this).delay(2000).queue(function() {
          $(this).removeClass('hover').dequeue();
        });
      });
      $(document).bind('click', function(e) {
        if($(e.target).closest('.shopping-cart-small').length === 0) {
           $('.shopping-cart-small').hide().removeClass('hover');
        }
      });
    });
  })(jQuery);
</script>
